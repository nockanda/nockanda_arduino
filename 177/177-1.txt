void loop() {
  if(digitalRead(refresh) == LOW){
    //풀업방식에서 LOW가 버튼이 눌려진것!
    tft.fillScreen(ILI9341_BLACK);
    RSS_READER();
  }
}

void RSS_READER(){
  // Use WiFiClientSecure class to create TLS connection
  WiFiClient client;
  Serial.print("connecting to ");
  Serial.println(host);

  //클라이언트와 서버와 커넥트
  if (!client.connect(host, httpsPort)) {
    Serial.println("connection failed");
    return;
  }

  String url = "/rss";
  Serial.print("requesting URL: ");
  Serial.println(url);

  //사물인터넷보드에서 RSS서버로 보내는 요청!
  client.print(String("GET ") + url + " HTTP/1.1\r\n" +
               "Host: " + host + "\r\n" +
               "Connection: close\r\n\r\n");

  //서버로부터 응답이 있을때까지 대기하겠다..
  //언제까지 기다려야하나요..?
  //timeout!
  unsigned long t = millis(); //생존시간
  while(1){
    if(client.available()) break;
    if(millis() - t > 10000) break;
  }


  //응답이 날라오는 부분!
  Serial.println("request sent");

  int count = 0;
  String text;
  while(client.available()){
    //String data = "<title> tes nockanda </title>";
    String data = client.readStringUntil('\n'); //readline();
    if(data.length() < 200){
      int s = data.indexOf("<title>"); //
      int e = data.lastIndexOf("</title>"); //
      if(s != -1 && e != -1){
       String output = data.substring(s+7,e);
       output = hangul_cut(output,250,16,10);
       //Serial.println(output); //output에 담긴 문자열을 PC에 시리얼 모니터에 출력해라~
       text += output + "\n";
       count++;
      }

      if(count >= 15) break; //테스트용
    }
  }
  //서버하고 클라이언트간 데이터 주고받기는 끝난상황!
  //이제 천천히 진행해도 OK!
  Serial.println("LCD에 프린트를 시작합니다");
  String line;
  int line_count=0;
  for(int i = 0;i<text.length();i++){
    if(text[i] == '\n'){
      //줄바꿈문자
      //할거하고~
      hangul2utf8(0,16*line_count,line,ILI9341_WHITE,16);
      //Serial.println(line);
      line = "";
      line_count++;
    }else{
      line += text[i];
    }
  }
  
  Serial.println("DONE");
}

//사이즈 몇으로 제한할래?
//한글폰트 사이즈 몇이냐?
//아스키코드 사이즈 몇이냐?
String hangul_cut(String input,int maxsize,int hangulsize,int asciisize){
  String output;
  int input_length = 0;
  for(int i = 0;i<input.length();){
    if(input[i] > 128){
      //한글(3바이트)
      output += input[i];
      output += input[i+1];
      output += input[i+2];
      input_length+=hangulsize;
      i+=3;
    }else{
      //아스키코드(1바이트)
      output += input[i];
      input_length += asciisize;
      i++;
    }

    if(input_length >= maxsize) break;
  }
  output += "..";
  return output;
}

void draw_hangul(int x,int y, String utf8code,uint16_t color,int font_size){
  
  int font_array_count = 0;
  String direct;

  if(font_size == 16) {font_array_count=2;direct=F("nock16/");}
  else if(font_size == 24) {font_array_count=3;direct=F("nock24/");}
  else if(font_size == 32) {font_array_count=4;direct=F("nock32/");}
  else if(font_size == 40) {font_array_count=5;direct=F("nock40/");}

  //파일불러오기
  myFile = SD.open(direct+utf8code+F(".txt"));
  String output;
  if (myFile) {
    while (myFile.available()) {
      output += String((char)myFile.read());
    }
    //Serial.println(output);
    myFile.close();
  } else {
    Serial.println("NO FILE");
    return;
  }

  int count = 1;
  //4개가 1line
  for(int i =0;i<output.length();i+=2){
    //Serial.println();
    //Serial.println(i);
    byte value = shex2hex(output[i], output[i+1]);

    for(int j =0;j<8;j++){
        byte temp = (128>>j);
        if((value & temp) == temp){
          //해당 bit가 1인것!
              tft.drawPixel(x,y,color);
              //Serial.print("1");
          }else{
              //Serial.print("0");
          }
          x++;
    }

   if(count % font_array_count ==0){
    //한라인구분
    //Serial.println();
    if(i == output.length()-2){
      y++;
      y-=font_size;
      
    }else{
      y++;
      x-=font_size;
    }
    //Serial.print("(");
    //Serial.print(i);
    //Serial.print(",");
    //Serial.print(x);
    //Serial.print(",");
    //Serial.print(y);
    //Serial.println(")");
   }
    count++;
  }//for
  tft.setCursor(x, y);
}

byte shex2hex(char data1, char data2){
//3F
//숫자는 -'0' 하면 숫자로된다
//a~f 는 -'a'+10 하면 숫자로된다
  byte output;

  int a,b;
  if(data1 >= '0' && data1 <= '9'){
   a = data1-'0';
  }else if(data1 >= 'a' && data1 <= 'f'){
   a = data1-'a'+10;
  }else if(data1 >= 'A' && data1 <= 'F'){
   a = data1-'A'+10; 
  }
  if(data2 >= '0' && data2 <= '9'){
   b = data2-'0';
  }else if(data2 >= 'a' && data2 <= 'f'){
   b = data2-'a'+10;
  }else if(data2 >= 'A' && data2 <= 'F'){
   b = data2-'A'+10; 
  }

  return a*16+b;

}

void hangul2utf8(int x,int y, String text,uint16_t color,int fontsize){
  tft.setTextColor(color); 
  if(fontsize == 16) tft.setTextSize(2);
  if(fontsize == 24) tft.setTextSize(3);
  
  tft.setCursor(x, y);
  for(int i = 0;i<text.length();){
    if(text[i] >128){
      //한글
      //Serial.println("한글");
      String output = String(text[i]&0xFF,HEX);
      output += String(text[i+1]&0xFF,HEX);
      output += String(text[i+2]&0xFF,HEX);
      i+=3;
      //Serial.println(output);
      //이전 커서값을 가져와야한다..
      
      draw_hangul(tft.getCursorX(),tft.getCursorY(),output,color,fontsize);

    }else{
      //ascii
      //Serial.println("아스키");
      //Serial.println(text[i]);
       tft.print(text[i]);
      i++;
    }

  }
}

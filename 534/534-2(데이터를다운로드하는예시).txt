#include <ESP8266WiFi.h>

#ifndef STASSID
#define STASSID "nockanda" //와이파이 검색했을때 뜨는 이름
#define STAPSK  "11213144" //패스워드
#endif

const char* ssid = STASSID;
const char* password = STAPSK;

//http://www.kma.go.kr/wid/queryDFSRSS.jsp?zone=1159068000
const char* host = "192.168.0.3";
const int httpsPort = 1880;

String API_KEY = "8107710c-2cc2-49f1-9059-7fef539d72de";

void setup() {
  Serial.begin(115200);
  Serial.println();
  Serial.print("connecting to ");
  Serial.println(ssid);
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.println("WiFi connected");
  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());
}

void loop() {
  Serial.println("값을 다운로드하겠습니다!");
  request();
  delay(5000);
}

void request(){
  WiFiClient client;
  Serial.print("connecting to ");
  Serial.println(host);
  
  //클라이언트와 서버와 커넥트
  if (!client.connect(host, httpsPort)) {
    Serial.println("connection failed");
    return;
  }

  String url = "/downdata?key="+API_KEY+"&num=10";
  Serial.print("requesting URL: ");
  Serial.println(url);

  //사물인터넷보드에서 RSS서버로 보내는 요청!
  client.print(String("GET ") + url + " HTTP/1.1\r\n" +
               "Host: " + host + "\r\n" +
               "Connection: close\r\n\r\n");

  //서버로부터 응답이 있을때까지 대기하겠다..
  //언제까지 기다려야하나요..?
  //timeout!
  unsigned long t = millis(); //생존시간
  while(1){
    if(client.available()) break;
    if(millis() - t > 10000) break;
  }

  //응답이 날라오는 부분!
  Serial.println("request sent");

  while(client.available()){
    String data = client.readStringUntil('\n'); //readline();
    Serial.println(data);
  }
  Serial.println("DONE");
}
